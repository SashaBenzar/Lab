#include <stdio.h>
#include <locale.h>
#include <math.h>

int main()
{
    int n,x,s1=0,s2=0,s3,i=1,j=1,a;
    printf("x=");
    scanf("%d",&x);
    printf("n=");
    scanf("%d",&n);
    while(i<=n)
    {
       s1+=pow(i,2);
       i++;
    }
    while(j<=n)
    {
       s2+=pow(j,2);
       j++;
    }
	s3=(x+i)/j;
    a=s1*s2*s3;
    printf("%d",a);
    return 0;
}
